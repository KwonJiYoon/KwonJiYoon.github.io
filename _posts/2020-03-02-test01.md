### 02. 두 정수 A와 B를 입력받고,A+B를 출력하는 프로그램을 작성하시오.(A > 0, B < 10)


~~~cpp
import java.util.*;

  public class Main {
    public static void main (String args[]){
      Scanner sc = new Scanner(System.in);
      //객체 sc는 시스템이 입력 받을 수 있는 새로운 Scanner
      
      int a,b;
      a = sc.nextInt();
      b = sc.nextInt();
      //sc에서 입력받은 값을 정수형으로 받음
      
      System.out.println(a+b);
    }
  }//Main
~~~
